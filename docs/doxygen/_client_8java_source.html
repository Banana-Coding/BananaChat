277  public boolean checkSpam(String channel) {
278  long now = System.currentTimeMillis();
279 
280  if (lastAction > now - 2000) {
281  spam++;
282 
283  if (spam == 2) {
284  sendButlerMessage(channel, "Bitte _NICHT spammen_ und fluten.");
285  } else if (spam == 3) {
286  disconnect();
287  return true;
288  }
289  } else {
290  spam -= (now - lastAction) / 2000;
291 
292  if (spam < 0) {
293  spam = 0;
294  }
295  }
296 
297  lastAction = now;
298  return false;
299  }

418  public void loadStats(String name) {
419  PoolConnection pcon = ConnectionPool.getConnection();
420  PreparedStatement ps = null;
421 
422  try {
423  Connection con = pcon.connect();
424  ps = con.prepareStatement("SELECT * FROM `accounts` WHERE `name` = ?");
425  ps.setString(1, name);
426  ResultSet rs = ps.executeQuery();
427 
428  if (rs.next()) {
429  this.name = rs.getString("name");
430  rank = rs.getByte("rank");
431 
432  icons = new ArrayList<Pair<String, Integer>>();
433  gender = rs.getByte("gender");
434 
435  if (gender == 1) {
436  addIcon("pics/male.png", 16);
437  } else if (gender == 2) {
438  addIcon("pics/female.png", 14);
439  }
440 
441  if (isVIP()) {
442  addIcon("pics/icon_vip.png", 22);
443  }
444 
445  kisses = rs.getInt("kisses");
446  knuddels = rs.getInt("knuddels");
447  onlineTime = rs.getInt("onlineTime");
448  loginTimestamp = System.currentTimeMillis();
449 
450  String[] registration = rs.getString("registration").split(" ");
451  String[] date = registration[0].split("-");
452  registrationDate = String.format("%s.%s.%s", date[2], date[1],
453  date[0]);
454  registrationTime = registration[1].split("\\.")[0];
455 
456  if (isModerator()) {
457  ipAddress = "1.3.3.7";
458  } else {
459  if (socket == null) {
460  ipAddress = rs.getString("ipAddress");
461  } else {
462  ipAddress = socket.getInetAddress().getHostAddress();
463  }
464  }
465 
466  away = false;
467  spam = 0;
468  lastAction = 0;
469  }
470  } catch (SQLException e) {
471  e.printStackTrace();
472  } finally {
473  if (ps != null) {
474  try {
475  ps.close();
476  } catch (SQLException e) {
477  }
478  }
479 
480  pcon.close();
481  }
482  }

535  public void sendPrivateMessage(List<Client> targets, String message,
536  Channel channel, boolean fromGame) {
537  if (!fromGame
538  && channel.getGame() != null
539  && !channel.getGame().parsePrivateMessage(targets, message,
540  this)) {
541  return;
542  }
543 
544  message = KCodeParser.parse(message, !isModerator(), 5, 10, 20);
545  message = Server.get().parseSmileys(message);
546 
547  if (message.isEmpty()) {
548  return;
549  }
550 
551  StringBuilder recipients = new StringBuilder();
552 
553  for (Client target : targets) {
554  recipients.append(target.getName());
555  recipients.append(",");
556  }
557 
558  String recipient = recipients.substring(0, recipients.length() - 1);
559 
560  if (!targets.contains(this)) {
561  targets.add(this);
562  }
563 
564  for (Client target : targets) {
565  if (channel.getClients().contains(target)) {
566  target.send(PacketCreator.privateMessage(name, recipient,
567  channel.getName(), message, " "));
568  } else {
569  for (Channel ch : target.getChannels()) {
570  target.send(PacketCreator.privateMessage(name, recipient,
571  ch.getName(), message, channel.getName()));
572  }
573  }
574  }
575  }
