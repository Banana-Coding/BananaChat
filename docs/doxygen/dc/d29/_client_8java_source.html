257  public boolean checkSpam(String channel) {

259 
260  if (lastAction > now - 2000) {

363  public void loadStats(String name) {
364  PoolConnection pcon = ConnectionPool.getConnection();
365  PreparedStatement ps = null;
366 
367  try {
368  Connection con = pcon.connect();
369  ps = con.prepareStatement("SELECT * FROM `accounts` WHERE `name` = ?");
370  ps.setString(1, name);
371  ResultSet rs = ps.executeQuery();
372 
373  if (rs.next()) {
374  this.name = rs.getString("name");
375  rank = rs.getByte("rank");
376 
377  icons = new ArrayList<Pair<String, Integer>>();
378  gender = rs.getByte("gender");
379 
380  if (gender == 1) {
381  addIcon("pics/male.png", 16);
382  } else if (gender == 2) {
383  addIcon("pics/female.png", 14);
384  }
385 
386  if (isVIP()) {
387  addIcon("pics/icon_vip.png", 22);
388  }
389 
390  kisses = rs.getInt("kisses");
391  knuddels = rs.getInt("knuddels");
392  onlineTime = rs.getInt("onlineTime");
393  loginTimestamp = System.currentTimeMillis();
394 
395  String[] registration = rs.getString("registration").split(" ");
396  String[] date = registration[0].split("-");
397  registrationDate = String.format("%s.%s.%s", date[2], date[1],
398  date[0]);
399  registrationTime = registration[1].split("\\.")[0];
400 
401  if (isModerator()) {
402  ipAddress = "1.3.3.7";
403  } else {
404  if (socket == null) {
405  ipAddress = rs.getString("ipAddress");
406  } else {
407  ipAddress = socket.getInetAddress().getHostAddress();
408  }
409  }
410 
411  away = false;
412  spam = 0;
413  lastAction = 0;
414  }
415  } catch (SQLException e) {
416  e.printStackTrace();
417  } finally {
418  if (ps != null) {
419  try {
420  ps.close();
421  } catch (SQLException e) {
422  }
423  }
424 
425  pcon.close();
426  }
427  }

463  public void sendPrivateMessage(List<Client> targets, String message,
464  Channel channel, boolean fromGame) {
465  if (!fromGame
466  && channel.getGame() != null
467  && !channel.getGame().parsePrivateMessage(targets, message,
468  this)) {
469  return;
470  }
471 
472  message = KCodeParser.parse(message, !isModerator(), 5, 10, 20);
473  message = Server.get().parseSmileys(message);
474 
475  if (message.isEmpty()) {
476  return;
477  }
478 
479  StringBuilder recipients = new StringBuilder();
480 
481  for (Client target : targets) {
482  recipients.append(target.getName());
483  recipients.append(",");
484  }
485 
486  String recipient = recipients.substring(0, recipients.length() - 1);
487 
488  if (!targets.contains(this)) {
489  targets.add(this);
490  }
491 
492  for (Client target : targets) {
493  if (channel.getClients().contains(target)) {
494  target.send(PacketCreator.privateMessage(name, recipient,
495  channel.getName(), message, " "));
496  } else {
497  for (Channel ch : target.getChannels()) {
498  target.send(PacketCreator.privateMessage(name, recipient,
499  ch.getName(), message, channel.getName()));
