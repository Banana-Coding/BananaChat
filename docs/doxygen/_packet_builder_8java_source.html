20 public class PacketBuilder {

31  public void write(int[] b) {
32  for (int i = 0; i < b.length; i++) {
33  writeByte(b[i]);
34  }
35  }
36 
37  public void writeByte(int b) {
38  buffer.append((char) b);
39  }
40 
41  public void writeShort(int s) {
42  writeByte((s >> 8) & 0xFF);
43  writeByte(s & 0xFF);
44  }
45 
46  public void writeString(String str) {

48  }

50  public void writeString(String str, boolean appendLength, boolean isShort) {
51  if (appendLength) {
52  int length = str.length();
53 
54  if (isShort) {
55  writeShort(length);
56  } else {
57  writeByte(length);
58  }
59  }
60 
61  buffer.append(str);
62  }
