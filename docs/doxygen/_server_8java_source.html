105  private void loadConfigs() {
106  PoolConnection pcon = ConnectionPool.getConnection();
107  Statement stmt = null;
108 
109  try {
110  Connection con = pcon.connect();
111  stmt = con.createStatement();
112  System.out.println("Loading butler");
113  ResultSet rs = stmt
114  .executeQuery("SELECT `name` FROM `accounts` ORDER BY `id` LIMIT 1");
115  rs.next();
116  butler.login(rs.getString("name"));
117  rs.close();
118  System.out.println("Loading channel styles");
119  rs = stmt.executeQuery("SELECT * FROM `channelstyles`");
120  Map<Integer, ChannelStyle> channelStyles = new HashMap<Integer, ChannelStyle>();
121 
122  while (rs.next()) {
123  channelStyles.put(rs.getInt("id"), new ChannelStyle(rs));
124  }
125 
126  rs.close();
127  System.out.println("Loading channels");
128  rs = stmt.executeQuery("SELECT * FROM `channels`");
129 
130  while (rs.next()) {
131  Channel channel = new Channel(rs, channelStyles.get(rs
132  .getInt("style")));
133  butler.joinChannel(channel);
134  channel.addClient(butler);
135  channels.put(rs.getString("name").toLowerCase(), channel);
136  }
137 
138  rs.close();
139  System.out.println("Loading smileys");
140  rs = stmt
141  .executeQuery("SELECT `code`, `replacement` FROM `smileys`");
142 
143  while (rs.next()) {
144  smileys.put(rs.getString("code"), rs.getString("replacement"));
145  }
146  } catch (SQLException e) {
147  e.printStackTrace();
148  } finally {
149  if (stmt != null) {
150  try {
151  stmt.close();
152  } catch (SQLException e) {
153  }
154  }
155 
156  pcon.close();
157  }
158  }
