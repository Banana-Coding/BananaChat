50  @Override
51  public String toString() {
52  PacketBuilder buffer = new PacketBuilder(SendOpcode.POPUP.getValue());
53 
54  buffer.writeByte(0x00);
55  addString(buffer, title);
56 
57  if (opcode != null) {
58  buffer.writeByte('s');
59  addString(buffer, opcode);
60  addString(buffer, parameter);
61  }
62 
63  addForeground(buffer, new int[] { 0x00, 0x00, 0x00 });
64  addBackground(buffer, new int[] { 0xBE, 0xBC, 0xFB });
65  buffer.writeByte(0xE3);
66 
67  // border right
68  buffer.writeByte('E');
69  buffer.writeByte('l');
70  addString(buffer, " ");
71  addFontStyle(buffer, 'p', 5);
72  addBackground(buffer, new int[] { 0xBE, 0xBC, 0xFB });
73  buffer.writeByte(0xE3);
74 
75  // border left
76  buffer.writeByte('W');
77  buffer.writeByte('l');
78  addString(buffer, " ");
79  addFontStyle(buffer, 'p', 5);
80  addBackground(buffer, new int[] { 0xBE, 0xBC, 0xFB });
81  buffer.writeByte(0xE3);
82 
83  buffer.writeByte('C');
84  addLayout(buffer, 'B');
85 
86  buffer.writeByte('N');
87  addLayout(buffer, 'B');
88 
89  // border top
90  buffer.writeByte('N');
91  buffer.writeByte('l');
92  addString(buffer, " ");
93  addFontStyle(buffer, 'p', 5);
94  addBackground(buffer, new int[] { 0xBE, 0xBC, 0xFB });
95  buffer.writeByte(0xE3);
96 
97  if (subtitle != null) {
98  buffer.writeByte('C');
99  buffer.writeByte('l');
100  addString(buffer, subtitle);
101  addFontStyle(buffer, 'b', 16);
102  addForeground(buffer, new int[] { 0x00, 0x00, 0x00 });
103  addBackground(buffer, new int[] { 0xE5, 0xE5, 0xFF });
104  buffer.writeByte(0xE3);
105 
106  buffer.writeByte('S');
107  buffer.writeByte('l');
108  addString(buffer, " ");
109  addFontStyle(buffer, 'p', 5);
110  addBackground(buffer, new int[] { 0xBE, 0xBC, 0xFB });
111  buffer.writeByte(0xE3);
112  }
113 
114  buffer.writeByte(0xE3);
115 
116  if (message != null) {
117  buffer.writeByte('C');
118  buffer.writeByte('c');
119  addString(buffer, message);
120  addFrameSize(buffer, width, height);
121  addForeground(buffer, new int[] { 0x00, 0x00, 0x00 });
122  addBackground(buffer, new int[] { 0xBE, 0xBC, 0xFB });
123  addBackgroundImage(buffer, "pics/cloudsblue.gif", 17);
124  buffer.writeByte(0xE3);
125  }
126 
127  boolean useBorderLayouts = panels.size() > 1;
128  int borderLayouts = 0;
129 
130  for (Panel panel : panels) {
131  buffer.writeByte('S');
132 
133  if (useBorderLayouts) {
134  borderLayouts++;
135  addLayout(buffer, 'B');
136  buffer.writeByte('N');
137  }
138 
139  addLayout(buffer, 'F');
140 
141  for (Component component : panel.getComponents()) {
142  ComponentType type = component.getType();
143  buffer.writeByte(type.getValue());
144 
145  if (type == ComponentType.CHECKBOX) {
146  if (component.getText() != null) {
147  buffer.writeByte('l');
148  addString(buffer, component.getText());
149  }
150 
151  addFontStyle(buffer, 'p', 16);
152  Checkbox checkbox = (Checkbox) component;
153 
154  if (checkbox.isDisabled()) {
155  buffer.writeByte('d');
156  }
157 
158  if (checkbox.isChecked()) {
159  buffer.writeByte('s');
160  buffer.writeByte('t');
161  }
162 
163  if (checkbox.getGroup() != 0) {
164  buffer.writeByte('r');
165  addSize(buffer, checkbox.getGroup());
166  }
167  } else {
168  addString(buffer, component.getText());
169 
170  if (type == ComponentType.BUTTON) {
171  addFontStyle(buffer, 'p', 16);
172  Button button = (Button) component;
173 
174  if (button.isStyled()) {
175  buffer.writeByte('c');
176 
177  if (button.isColored()) {
178  buffer.writeByte('e');
179  }
180  }
181 
182  if (button.isClose()) {
183  buffer.writeByte('d');
184  }
185 
186  if (button.isAction()) {
187  buffer.writeByte('s');
188  }
189 
190  if (button.getCommand() != null) {
191  buffer.writeByte('u');
192  addString(buffer, button.getCommand());
193  }
194  } else if (type == ComponentType.TEXT_FIELD) {
195  addSize(buffer, ((TextField) component).getWidth());
196  } else if (type == ComponentType.LABEL) {
197  addFontStyle(buffer, 'p', 16);
198  } else if (type == ComponentType.TEXT_AREA) {
199  TextArea textarea = (TextArea) component;
200  addSize(buffer, textarea.getRows());
201  addSize(buffer, textarea.getColumns());
202 
203  switch (textarea.getScrollbars()) {
204  case 0:
205  buffer.writeByte('b');
206  break;
207  case 1:
208  buffer.writeByte('s');
209  break;
210  case 2:
211  buffer.writeByte('w');
212  break;
213  }
214 
215  if (textarea.isEditable()) {
216  buffer.writeByte('e');
217  }
218  }
219  }
220 
221  addForeground(buffer, component.getForeground());
222  addBackground(buffer, component.getBackground());
223  buffer.writeByte(0xE3);
224  }
225 
226  buffer.writeByte(0xE3);
227  }
228 
229  for (int i = 0; i < borderLayouts; i++) {
230  buffer.writeByte(0xE3);
231  }
232 
233  buffer.writeByte(0xE3);
234  buffer.writeByte(0xE3);
235 
236  return buffer.toString();
237  }

244  buffer.writeString(str);

248  private static void addFontStyle(PacketBuilder buffer, char weight, int size) {
249  if (weight != 'p') {

252 
253  buffer.writeByte('g');
254  addSize(buffer, size);
255  }

257  private static void addLayout(PacketBuilder buffer, char layout) {
258  buffer.writeByte('p');
259  buffer.writeByte(layout);

262  private static void addFrameSize(PacketBuilder buffer, int width, int height) {

266  }

268  private static void addForeground(PacketBuilder buffer, int[] color) {
269  buffer.writeByte('f');
270  buffer.write(color);
271  }
272 
273  private static void addBackground(PacketBuilder buffer, int[] color) {
274  buffer.writeByte('h');
275  buffer.write(color);
276  }
277 
278  private static void addBackgroundImage(PacketBuilder buffer, String image,
279  int position) {
280  buffer.writeByte('i');
281  addString(buffer, image);
282  buffer.writeShort(position);
283  }
