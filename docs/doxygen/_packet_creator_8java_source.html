165  public static String channelList(Collection<Channel> channels) {
166  PacketBuilder buffer = new PacketBuilder(
167  SendOpcode.CHANNEL_LIST.getValue());
168 
169  for (Channel channel : channels) {
170  int count = channel.countClients();
171  buffer.writeByte(0x00);
172  buffer.writeString(channel.getName());
173  buffer.writeByte(0x0A);
174  buffer.writeString(String.valueOf(count));
175  buffer.writeByte(0x00);
176  buffer.writeByte('p'); // font weight
177  buffer.writeByte(0x00);
178  buffer.writeString("B");
179 
180  if (count == channel.getSize()) {
181  buffer.writeByte(0x00);
182  buffer.writeString("pics/icon_fullChannel.gif");
183  }
184 
185  buffer.writeByte(0x00);
186  buffer.writeString("-");
187  }
188 
189  return buffer.toString();
190  }

233  public static String addUser(Channel channel, Client client, boolean self) {
234  PacketBuilder buffer = new PacketBuilder(SendOpcode.ADD_USER.getValue());
235 
236  buffer.writeByte(0x00);
237  buffer.writeString(channel.getName());
238  buffer.writeByte(0x00);
239  buffer.writeString(client.getName());
240  buffer.writeByte(0x00);
241  buffer.writeByte(self ? 'b' : 'p'); // font weight
242  buffer.writeByte(0x00);
243  buffer.writeString(channel.getStyle().getRankColor(client));
244 
245  for (Pair<String, Integer> icon : client.getIcons()) {
246  buffer.writeByte(0x00);
247  buffer.writeString(icon.getLeft());
248  buffer.writeByte(0x00);
249  buffer.writeString(String.valueOf(icon.getRight()));
250  }
251 
252  buffer.writeByte(0x00);
253  buffer.writeString("-"); // login message
254 
255  return buffer.toString();
256  }
