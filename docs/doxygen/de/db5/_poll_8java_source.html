136  public void vote(Client client, String answer) {
137  synchronized (votes) {
138  if (!votes.containsKey(client.getName())
139  && answers.contains(answer)) {
140  if (closed) {
141  String popup = Popup
142  .create("Umfrage",
143  String.format("Umfrage #%s - Problem", id),
144  "Du hast dir für die Beantwortung der Umfrage zu lange Zeit gelassen.",
145  400, 300);
146  client.send(popup);
147  } else {
148  votes.put(client.getName(), answer);
149  }
150  }
